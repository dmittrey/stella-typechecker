language core;
extend with #variants,#unit-type,#type-ascriptions,#natural-literals,#nullary-variant-labels;

fn example(x : Bool) -> <| a : Nat, b |> {
  return if x then <| a = 5 |> else <| b |>
}

fn main(flag : Bool) -> Nat {
  return match example(flag) {
      <| a = n |> => succ(n)
    | <| b = n |> => 0
  }
}
